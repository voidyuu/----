typedef struct node{
 char data[10];
 struct node *left,*right;
}BTree;

void in_traverse(BTree* node,string s){
 if(node!=NULL){
  in_traverse(node->left,s);
  if(node->left!=NULL ||node->right!=NULL){
   s+="("+node->data+")";
  }else{
   s+=node->data;
  }
  in_traverse(node->right,s);
 }
}

string to_expression(BTree* node){
 string s="";
 in_traverse(node,s);
 return s;
}
