将一维数组 R 中的序列循环左移 p 个位置。

```cpp
void reverse(int * arr,int start,int len){
 int tmp[len];
 for(int i=0;i<len;i++){
  tmp[i]=arr[start+len-1-i];
 }
 for(int i=start;i<start+len;i++){
  arr[i]=tmp[i-start];
 }
}

void Converse(int * R,int p,int n){
 reverse(R,0,n);
 reverse(R,0,n-p);
 reverse(R,n-p,p);
}
```
