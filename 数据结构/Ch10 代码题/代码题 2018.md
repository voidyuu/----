长度为 n 的数组，求出其中未出现的最小正整数

```cpp
int find_miss_min(int A[],int n){
 int i,*B;//标记数组
 B=(int *) malloc(sizeof(int)*n);//分配空间
 memset(B,0,sizeof(int)*n);//赋初值为0
 for(i=0;i<n;i++){
  if(A[i]>0 && A[i]<=n){
   B[A[i]-1]=1;//若A[i]的值介于1~n，则标记数组B
  }
 }
 for(i=0;i<n;i++){
  if(B[i]==0)break;//扫描数组B，找到目标值
 }
 return i+1;//返回结果
}
```
