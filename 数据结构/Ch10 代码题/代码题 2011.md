int find_mid_number(int a[],int b[],int L){
 int i=0;
 int j=0;
 for(int len=0;len<L-1;len++){
  if(a[i]<=b[j]){
   i++;
  }else{
   j++;
  }
 }
 int mid=a[i]<=b[j]?a[i]:b[j];
 return mid;
}
